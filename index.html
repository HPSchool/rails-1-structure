2. Installing rbenv and ruby build (on Mac)

> brew install rbenv ruby build

> echo 'eval "$(rbenv init -)"' >> ~/.bash_profile

# If you use zsh (can also be replaced with .zsh_profile)

installation

2. Installing rbenv and ruby build (on Linux)

> git clone https://github.com/sstephenson/rbenv.git ~/.rbenv
> git clone https://github.com/sstephenson/ruby-build.git ~/.rbenv/plugins/ruby-build

> echo 'export PATH="$HOME/.rbenv/bin:$PATH"' >> ~/.bashrc
> echo 'eval "$(rbenv init -)"' >> ~/.bashrc

# If you use zsh replace .bashrc with .zshrc
# If you use your linux as server replace with .bash_profile

close your terminal after this and reopen it or issue exec $SHELL -l
